import pygame import random import math # 1. تهيئة اللعبة والمصادر الصوتية pygame.init() pygame.mixer.init() # أبعاد الشاشة SCREEN_WIDTH = 800 SCREEN_HEIGHT = 600 screen = pygame.display.set_mode((SCREEN_WIDTH, SCREEN_HEIGHT)) pygame.display.set_caption("🛡️ حرب الطائرات الحماسية 🛡️") # الألوان WHITE = (255, 255, 255) BLACK = (0, 0, 0) RED = (255, 0, 0) GREEN = (0, 255, 0) YELLOW = (255, 255, 0) # تحميل الأصوات (تأكد من وجود الملفات في نفس المجلد) try: pygame.mixer.music.load('background.mp3') # موسيقى الخلفية pygame.mixer.music.play(-1) # -1 تعني تكرار مستمر laser_sound = pygame.mixer.Sound('laser.wav') # صوت الإطلاق explosion_sound = pygame.mixer.Sound('explosion.wav') # صوت الانفجار except: print("تنبيه: لم يتم العثور على ملفات الصوت، ستعمل اللعبة بدون صوت.") laser_sound = None explosion_sound = None # ساعة اللعبة للتحكم بالـ FPS clock = pygame.time.Clock() # --- عناصر اللعبة --- # 1. طائرة اللاعب player_width = 60 player_height = 50 player_x = SCREEN_WIDTH // 2 player_y = SCREEN_HEIGHT - 70 player_speed = 8 # 2. الأعداء (طائرات العدو) enemy_width = 50 enemy_height = 40 num_of_enemies = 6 enemies = [] def spawn_enemy(): return { "x": random.randint(0, SCREEN_WIDTH - enemy_width), "y": random.randint(-150, -50), "speed": random.randint(3, 6) } for i in range(num_of_enemies): enemies.append(spawn_enemy()) # 3. الصواريخ bullet_width = 5 bullet_height = 15 bullet_speed = 12 bullets = [] # النتيجة والقلوب score = 0 lives = 3 font = pygame.font.SysFont("Arial", 30) game_over_font = pygame.font.SysFont("Arial", 60) def show_score_and_lives(): score_text = font.render(f"Score: {score}", True, WHITE) lives_text = font.render(f"Lives: {'❤️' * lives}", True, RED) screen.blit(score_text, (10, 10)) screen.blit(lives_text, (SCREEN_WIDTH - 150, 10)) # دالة التحقق من التصادم (المسافة بين جسمين) def is_collision(x1, y1, x2, y2, distance_threshold): distance = math.sqrt((x1 - x2)*2 + (y1 - y2)*2) return distance < distance_threshold # --- الحلقة الرئيسية للعبة --- running = True game_over = False while running: screen.fill(BLACK) # خلفية الفضاء السوداء for event in pygame.event.get(): if event.type == pygame.QUIT: running = False # إطلاق الصواريخ عند الضغط على زر Space if event.type == pygame.KEYDOWN: if event.key == pygame.K_SPACE and not game_over: bullets.append({"x": player_x + player_width//2 - bullet_width//2, "y": player_y}) if laser_sound: laser_sound.play() # إعادة اللعب إذا انتهت if event.key == pygame.K_r and game_over: game_over = False score = 0 lives = 3 enemies = [spawn_enemy() for _ in range(num_of_enemies)] bullets.clear() pygame.mixer.music.play(-1) if not game_over: # حركة اللاعب (أزرار الأسهم يمين ويسار) keys = pygame.key.get_pressed() if keys[pygame.K_LEFT] and player_x > 0: player_x -= player_speed if keys[pygame.K_RIGHT] and player_x < SCREEN_WIDTH - player_width: player_x += player_speed # تحريك وتحديث الصواريخ for bullet in bullets[:]: bullet["y"] -= bullet_speed if bullet["y"] < 0: bullets.remove(bullet) # حذف الصاروخ إذا خرج من الشاشة # تحريك وتحديث الأعداء for enemy in enemies: enemy["y"] += enemy["speed"] # إذا مر العدو دون تدميره تخسر محاولة if enemy["y"] > SCREEN_HEIGHT: lives -= 1 enemies.remove(enemy) enemies.append(spawn_enemy()) if lives <= 0: game_over = True # تحقق من تصادم الصاروخ مع العدو for bullet in bullets[:]: if is_collision(enemy["x"] + enemy_width//2, enemy["y"] + enemy_height//2, bullet["x"] + bullet_width//2, bullet["y"] + bullet_height//2, 30): if explosion_sound: explosion_sound.play() score += 10 bullets.remove(bullet) enemies.remove(enemy) enemies.append(spawn_enemy()) break # تحقق من تصادم طائرة العدو مع طائرة اللاعب if is_collision(enemy["x"] + enemy_width//2, enemy["y"] + enemy_height//2, player_x + player_width//2, player_y + player_height//2, 40): if explosion_sound: explosion_sound.play() lives -= 1 enemies.remove(enemy) enemies.append(spawn_enemy()) if lives <= 0: game_over = True # --- الرسم على الشاشة --- # رسم طائرة اللاعب (مثلث أخضر حربي) pygame.draw.polygon(screen, GREEN, [ (player_x + player_width//2, player_y), (player_x, player_y + player_height), (player_x + player_width, player_y + player_height) ]) # رسم الأعداء (مربعات حمراء غاضبة) for enemy in enemies: pygame.draw.rect(screen, RED, (enemy["x"], enemy["y"], enemy_width, enemy_height)) # رسم الصواريخ (خطوط صفراء مضيئة) for bullet in bullets: pygame.draw.rect(screen, YELLOW, (bullet["x"], bullet["y"], bullet_width, bullet_height)) show_score_and_lives() else: # شاشة النهاية Game Over pygame.mixer.music.stop() go_text = game_over_font.render("GAME OVER", True, RED) restart_text = font.render("Press 'R' to Restart or Close Window", True, WHITE) final_score = font.render(f"Your Score: {score}", True, GREEN) screen.blit(go_text, (SCREEN_WIDTH//2 - 150, SCREEN_HEIGHT//2 - 80)) screen.blit(final_score, (SCREEN_WIDTH//2 - 80, SCREEN_HEIGHT//2)) screen.blit(restart_text, (SCREEN_WIDTH//2 - 200, SCREEN_HEIGHT//2 + 60)) pygame.display.update() clock.tick(60) # 60 إطار في الثانية للحفاظ على سلاسة وحماس اللعبة pygame.quit()
